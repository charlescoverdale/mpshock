% Extra preamble for the R Journal rendering
\usepackage{longtable}

% Stop figures floating into the bibliography.
\usepackage{placeins}

% Allow figures to float to cleaner positions (top/bottom of nearby
% pages) rather than leaving whitespace at hard [H] placement. Tune
% LaTeX's float parameters so we can put a bigger fraction of a page
% into a single figure before the algorithm gives up and leaves gaps.
\usepackage{float}
\renewcommand{\topfraction}{0.92}
\renewcommand{\bottomfraction}{0.8}
\renewcommand{\floatpagefraction}{0.7}
\renewcommand{\textfraction}{0.07}
\setcounter{topnumber}{3}
\setcounter{bottomnumber}{2}
\setcounter{totalnumber}{4}

% Suppress the auto-generated \address{...} trailer.
\renewcommand{\address}[1]{}

% Number sections and subsections (match recent R Journal convention).
% pandoc maps ## to \subsection, which RJournal.sty leaves unnumbered by
% default. Override the titleformat so subsections display a single
% arabic counter ("1 Introduction", "2 Background", ...).
\setcounter{secnumdepth}{3}
\usepackage{titlesec}
\renewcommand\thesubsection{\arabic{subsection}}
\renewcommand\thesubsubsection{\arabic{subsection}.\arabic{subsubsection}}
\titleformat{\subsection}{\normalfont\large\bfseries}{\thesubsection}{1em}{}
\titleformat{\subsubsection}{\normalfont\normalsize\bfseries}{\thesubsubsection}{1em}{}

% Blue DOI / URL / citation hyperlinks (RJournal.sty leaves colorlinks off).
\hypersetup{colorlinks=true,
  linkcolor=blue!60!black,
  urlcolor=blue!60!black,
  citecolor=blue!60!black}

% Make DOIs in the bibliography clickable. natbib / plainnat prints
% "doi: 10.xxxx" as plain text; redefine \doi so the string is wrapped
% in \href and gets the blue colorlinks treatment.
\renewcommand{\doi}[1]{\href{https://doi.org/#1}{doi:~#1}}
